28// HACKY WAY TO DETERMINE SCALAR TYPE AS YAML-CPP DOESN'T OFFER THAT..
29namespace YAML {
30template <typename T> struct as_if<T, std::optional<T>> {
31 explicit as_if(const Node &input_node) : node(input_node) {}
32 const Node &node;
33
34 const std::optional<T> operator()() const {
35 std::optional<T> val;
36 T t;
37 if (node.m_pNode && convert<T>::decode(node, t))
38 val = std::move(t);
39
40 return val;
41 }
42};
43
44// There is already a std::string partial specialisation, so we need a full
45// specialisation here
46template <> struct as_if<std::string, std::optional<std::string>> {
47 explicit as_if(const Node &input_node) : node(input_node) {}
48 const Node &node;
49
50 const std::optional<std::string> operator()() const {
51 std::optional<std::string> val;
52 std::string t;
53 if (node.m_pNode && convert<std::string>::decode(node, t))
54 val = std::move(t);
55
56 return val;
57 }
58};
59} // namespace YAML
